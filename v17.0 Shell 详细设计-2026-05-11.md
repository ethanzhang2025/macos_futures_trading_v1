## 3 · 核心抽象：PaneProtocol

### 3.1 协议定义

```swift
// Sources/MainApp/Shell/PaneProtocol.swift

import SwiftUI

/// Pane 协议：所有可嵌入 Shell 的 view 实现
protocol PaneProtocol {
    /// 唯一 ID（用于 group binding / 序列化）
    var paneID: UUID { get }

    /// Pane 类型
    var paneKind: PaneKind { get }

    /// 当前绑定的合约 symbol（nil = 未绑定）
    var paneSymbol: String? { get set }

    /// 当前绑定的周期（nil = 不适用 / 跨周期 group）
    var panePeriod: KLinePeriod? { get set }

    /// 当前 group 颜色（nil = 不参与联动）
    var paneGroupColor: GroupColor? { get set }

    /// 视图渲染
    func paneView() -> AnyView

    /// 工具栏（可空）
    func paneToolbar() -> AnyView?

    /// 序列化为 Data（持久化用 · 默认空 Data）
    func encodeExtra() -> Data?

    /// 从 Data 反序列化（持久化恢复用）
    mutating func decodeExtra(_ data: Data)
}
```

### 4.4 PaneContainer 实现骨架

```swift
struct PaneContainer: View {
    let workspace: Workspace
    @EnvironmentObject var shellVM: ShellViewModel

    var body: some View {
        switch workspace.paneLayout {
        case .single:
            paneFor(workspace.panes[0])
        case .twoVertical:
            VSplitView {
                paneFor(workspace.panes[0])
                paneFor(workspace.panes[1])
            }
        case .twoHorizontal:
            HSplitView {
                paneFor(workspace.panes[0])
                paneFor(workspace.panes[1])
            }
        case .four:
            VSplitView {
                HSplitView { paneFor(0); paneFor(1) }
                HSplitView { paneFor(2); paneFor(3) }
            }
        // ... sixGrid / nineGrid
        }
    }

    @ViewBuilder
    private func paneFor(_ config: PaneConfig) -> some View {
        PaneHost(config: config)
            .environment(\.isHostedInShell, true)
    }
}

struct PaneHost: View {
    let config: PaneConfig
    var body: some View {
        VStack(spacing: 0) {
            PaneHeader(config: config)  // group 色 chip + symbol picker + 关闭 button
            PaneBody(config: config)    // 实际嵌入对应 view
        }
    }
}

struct PaneBody: View {
    let config: PaneConfig
    var body: some View {
        switch config.kind {
        case .chart: ChartScene()
        case .spread: SpreadWindow()
        case .option: OptionWindow()
        case .review: ReviewWindow()
        case .training: TrainingWindow()
        // ... 其他 PaneKind 映射到对应 view
        }
    }
}
```

### 5.3 ShellViewModel 联动逻辑

```swift
@MainActor
final class ShellViewModel: ObservableObject {
    @Published var groupBindings: [GroupColor: SymbolBinding] = [:]

    /// Pane 设定 group 色 → 同步当前 binding 到 Pane
    func bindPaneToGroup(_ paneID: UUID, color: GroupColor) {
        // ...
    }

    /// Pane 改 symbol → 广播到同组所有 Pane
    func paneDidChangeSymbol(_ paneID: UUID, symbol: String) {
        guard let color = groupColorFor(paneID) else { return }
        groupBindings[color]?.symbol = symbol
        // 自动触发同组 Pane 的 @Published 更新
    }

    /// 鼠标移到 Pane（k 线某 bar）→ 广播十字光标时间
    func paneDidHoverTime(_ paneID: UUID, time: Date) {
        guard let color = groupColorFor(paneID) else { return }
        groupBindings[color]?.crosshairTime = time
    }
}
```

## 9 · 状态管理 · 持久化 · 启动恢复

### 9.1 启动流程

```
App 启动
  ↓
读取 @AppStorage "shell.workspaces" JSON
  ↓
若为空：加载 Workspace.defaults（白天/夜盘/复盘 3 个）
若非空：还原用户上次的 workspaces 列表
  ↓
读取 @AppStorage "shell.activeWorkspaceID"
  ↓
打开 Shell 主窗口 · 默认切到 active workspace
  ↓
PaneContainer 渲染对应 PaneLayout + Panes
```

### 9.2 持久化点

| 键 | 内容 | 触发时机 |
|----|------|---------|
| `shell.workspaces` | `[Workspace]` JSON | 每次 workspace 增删改时 |
| `shell.activeWorkspaceID` | UUID | 切 Tab 时 |
| `shell.primaryTab` | PrimaryTab.rawValue | 切一级模块时 |
| `shell.sidebarCollapsed` | Bool | 折叠 sidebar 时 |
| `shell.inspectorVisible` | Bool | 显示/隐藏 inspector 时 |
| `shell.bottomBarCollapsed` | Bool | 折叠底部交易区时 |
| `shell.groupBindings` | `[GroupColor: SymbolBinding]` | symbol/period 变化时 |

### 9.3 ⌘+Q 保留状态（TV 风）

- App 退出时持久化所有状态
- 下次启动**完全恢复**（Tab / Pane / 滚动位置 / 十字光标 / sidebar 折叠等）
