### 6.7 parseHistoricalDate · `en_US_POSIX` locale 关键 bug 修复

private static func parseHistoricalDate(_ s: String) -> Date? {

    f.locale = Locale(identifier: "en_US_POSIX")    // ← 用户系统中文 locale 时不加这个会全失败

}
