### 5.5 batch208 修 flaky · async timing 根因揭示

```swift
// 测试模式（之前 stress 下 100% flaky）：
let observeTask = Task {
    for await event in await engine.observe() {  // ← AsyncStream builder closure 是 deferred！
        ...
    }
}
// ← 主线程立即开始 onTick · observeTask 还没进入 for await 状态
await engine.onTick(...)  // ← event 推到 continuations dict · 但还是空的
                          //    （observeTask 此时还没执行 stream builder，没注册 continuation）
                          //    → event 丢失

// 修复：waitForObserveReady() helper
private func waitForObserveReady() async {
    await Task.yield()                                  // 让出 cooperative thread
    try? await Task.sleep(nanoseconds: 30_000_000)     // 30ms 起步窗口
}
// 在 observeTask 创建后立即调用 · 让 task 进入 for await 状态再开始 onTick

// 配合 waitForCount polling helper（替代固定 sleep · 5ms 间隔 · 1.5s timeout）
@discardableResult
private func waitForCount(_ collector, target: Int, timeout: Double = 1.5) async -> Int { ... }
```
